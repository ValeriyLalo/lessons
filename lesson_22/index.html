Задача №4

Створіть функцію, яка повертає результат ділення числа a на число b з додаванням рядка "Результат ділення: " Викличте функцію передаючі різні значення, у тому числі не передаючи зовсім. Функція не має повертати NaN, Infinite або помилку

Рішення:
function calculation(a = 0, b = 0) {
if(a !== 0 || b !== 0) {
a >= 1
b >= 1
}
if ( b === 0 ) {
return `Результат ділення:на нуль ділити не можна! `
}
if ( a === 0 ) {
return `Результат ділення:0`
}
if (typeof (a) !== `number` || typeof (b) !== `number`){
return `Результат ділення: дільник не є числом!`
}
let result = a / b
return`Результат ділення: ${result}`
}
console.log(calculation(10, 2)) //Результат ділення:5
console.log(calculation(30, 3)) //Результат ділення:10
console.log(calculation(30,0)) //Результат ділення:на нуль ділити не можна!
console.log(calculation(0,20)) //Результат ділення:0
console.log(calculation(56, 8)) //Результат ділення:7
console.log(calculation("Коля", 8)) //Результат ділення: дільник не є числом!
console.log(calculation(-10, 2)) //Результат ділення:-5
